## 三、判断输出（20 分）

1. ```c
    #include<stdio.h>
    int main() {
      int row = 3, col = 4, i, j;
      for(i = 0; i < row; i++) {
        printf("#");
        for(j = 0; j < col; j ++) {
          if(i == j) break;
          if(i == col - j) continue;
          printf("*");
        }
        printf("\n");
      }
      return 0;
    }
    ```

2. ```c
    #include<stdio.h>
    #define SQUARE(x) x*x
    int main() {
      int a = 5;
      int b = SQUARE(a + 1);
      printf("SQUARE(a+1):%d",b);
      return 0;
    }
    ```
