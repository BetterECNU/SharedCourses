2. （题目暂缺）

    <details>
    <summary>答案：</summary>

    100 200 301 300

    `*p1 = 100`，`*p2 = 100`，`*p3 = 300`。`printf("%d %d %d %d", *p1++, *++p2, *p3, (*p3)++)` 语句从后往前运算。`(*p3)++` 为后 `++`，所以输出值为 300，指向的内存空间值被更改为 301。`*p3` 的值为 301。`*++p2` 先对 `p2++`，此时 `p2` 指向 `data` 数组的第二个元素，值为 200，所以输出为 200。`*p1++` 先输出后 `++`，因此输出值为 100。

    </details>

    ***

4. （题目暂缺）

    <details>
    <summary>答案：</summary>

    144

    递归运算是将程序压入栈（先进后出的数据结构）。`fun(81)` 先入栈，调用 `f(11)`。`f(11)` 入栈，调用 `f(1)`。`f(1)` 入栈，调用 `f(0)`。`f(0)` 入栈，因为 `n` 不符合条件，函数结束。`f(0)` 出栈，继续执行调用 `f(0)` 的 `f(1)`，输出 1。`f(1)` 出栈，执行 `f(11)` 的后续语句，输出 4，函数结束，`f(11)` 出栈。执行 `f(81)` 后续语句，输出 4，`f(81)` 出栈。程序调用结束。

    </details>

### 三、编程题

1. （题目暂缺）

    <details>
    <summary>答案：</summary>

    ```c
    #include <stdio.h>
    #include <stdlib.h>
    #include <string.h>

    // 定义结构体来存储书籍信息
    typedef struct Book {
        int score;
        char author[100];
        char title[100];
    } Book;

    int compare(const void *a, const void *b) {
        Book *bookA = (Book *)a;
        Book *bookB = (Book *)b;

        // 先按照评分从高到低排序
        if (bookA->score > bookB->score) {
            return -1;
        } else if (bookA->score < bookB->score) {
            return 1;
        } else {
            // 评分相同，按照作者名字典序排序
            int authorCmp = strcmp(bookA->author, bookB->author);
            if (authorCmp != 0) {
                return authorCmp;
            } else {
                // 作者名相同，按照书名字典序排序
                return strcmp(bookA->title, bookB->title);
            }
        }
    }

    int main() {
        int n;
        scanf("%d", &n);

        Book books[n];
        for (int i = 0; i < n; i++) {
            scanf("%d %s %s", &books[i].score, books[i].author, books[i].title);
        }

        qsort(books, n, sizeof(Book), compare);

        for (int i = 0; i < n; i++) {
            printf("%d %s %s\n", books[i].score, books[i].author, books[i].title);
        }

        return 0;
    }
    ```

    </details>

2. （题目暂缺）

    <details>
    <summary>答案：</summary>

    ```c
    #include <stdio.h>

    int main() {
        int nums[7] = {1, 2, 3, 4, 5, 6, 7};
        int k;
        scanf("%d", &k);
        k %= 7;
        int newNums[7];

        // 将 nums 数组中从索引为(n - k)的元素开始，依次放入 newNums 数组的开头
        for (int i = 0; i < k; i++) {
            newNums[i] = nums[7 - k + i];
        }

        // 将 nums 数组中从索引为 0 到(n - k - 1)的元素，依次放入 newNums 数组中剩余的位置
        for (int i = 0; i < 7 - k; i++) {
            newNums[k + i] = nums[i];
        }

        // 输出轮转后的数组
        for (int i = 0; i < 7; i++) {
            printf("%d ", newNums[i]);
        }

        return 0;
    }
    ```

    </details>
